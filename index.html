mapParallel <- function(.data, .function, clusters = 2 ){
  		require( parallel )
  		require( purrr )
  
  		cl <- makeCluster( getOption( "cl.cores", clusters ))
  		temp <-  unlist( clusterCall(cl = cl, fun = map, .x = .data, .f = .function), recursive = FALSE )
  		stopCluster(cl)
  
  		return( temp )
	} # END mapParallel
